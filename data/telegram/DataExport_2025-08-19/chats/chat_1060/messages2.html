### 1. Smart Contract in Solidity

This Solidity contract will handle depositing BTC, providing liquidity, opening perpetual positions, and rebalancing between dlcBTC and PerpBTC.

#### Vault.sol

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@enzymefinance/contracts/EnzymeVault.sol"; // Placeholder for Enzyme's vault contract

interface IDex {
    function addLiquidity(uint256 amount) external;
    function removeLiquidity(uint256 amount) external;
}

interface IPerpetual {
    function openPosition(uint256 amount, uint256 leverage) external;
    function closePosition(uint256 amount) external;
}

contract BTCVault {
    address public owner;
    IERC20 public btcToken;
    IDex public dex;
    IPerpetual public perpetual;
    EnzymeVault public enzymeVault;
    uint256 public dlcBTCAllocation; // Percentage allocated to dlcBTC
    uint256 public perpBTCAllocation; // Percentage allocated to PerpBTC

    constructor(
        address _btcToken,
        address _dex,
        address _perpetual,
        address _enzymeVault,
        uint256 _dlcBTCAllocation,
        uint256 _perpBTCAllocation
    ) {
        owner = msg.sender;
        btcToken = IERC20(_btcToken);
        dex = IDex(_dex);
        perpetual = IPerpetual(_perpetual);
        enzymeVault = EnzymeVault(_enzymeVault);
        dlcBTCAllocation = _dlcBTCAllocation;
        perpBTCAllocation = _perpBTCAllocation;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function");
        _;
    }

    function deposit(uint256 amount) external onlyOwner {
        btcToken.transferFrom(msg.sender, address(this), amount);
        rebalance();
    }

    function rebalance() public onlyOwner {
        uint256 balance = btcToken.balanceOf(address(this));
        uint256 dlcAmount = (balance * dlcBTCAllocation) / 100;
        uint256 perpAmount = (balance * perpBTCAllocation) / 100;

        // Provide liquidity
        dex.addLiquidity(dlcAmount);

        // Open perpetual position
        perpetual.openPosition(perpAmount, 2); // Example leverage of 2x
    }

    function withdraw(uint256 amount) external onlyOwner {
        // Remove liquidity and close positions proportionally
        uint256 dlcAmount = (amount * dlcBTCAllocation) / 100;
        uint256 perpAmount = (amount * perpBTCAllocation) / 100;

        dex.removeLiquidity(dlcAmount);
        perpetual.closePosition(perpAmount);

        btcToken.transfer(msg.sender, amount);
    }
}
